## Anatomy

Import the components and place them together:

```jsx title="Anatomy"
import { FilterableMenu } from '@base-ui-components/react/filterable-menu';

<FilterableMenu.Root>
  <FilterableMenu.Trigger />

  <FilterableMenu.Portal>
    <FilterableMenu.Positioner>
      <FilterableMenu.Popup>
        <FilterableMenu.Status />
        <FilterableMenu.Empty />

        <FilterableMenu.List>
          <FilterableMenu.Row>
            <FilterableMenu.Item />
          </FilterableMenu.Row>

          <FilterableMenu.Separator />

          <FilterableMenu.Group>
            <FilterableMenu.GroupLabel />
          </FilterableMenu.Group>

          <FilterableMenu.Collection />
        </FilterableMenu.List>
      </FilterableMenu.Popup>
    </FilterableMenu.Positioner>
  </FilterableMenu.Portal>
</FilterableMenu.Root>;
```
